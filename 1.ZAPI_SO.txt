define view entity ZI_SO_COUNT_STATUS
  with parameters
    customer : kunnr

  as select from I_SalesDocument
{
  key count( distinct case OverallSDProcessStatus
                       when 'A' then SalesDocument end )      as Notprocessed,
      count( distinct case OverallSDProcessStatus
                            when 'B' then SalesDocument end ) as Partiallyprocessed,
      count( distinct case OverallSDProcessStatus
                           when 'C' then SalesDocument end )  as Completelyprocessed,
      count( distinct case OverallSDProcessStatus
                           when ' ' then SalesDocument end )  as Notrelevant


}
where
  SoldToParty = lpad( $parameters.customer , 10 , '0' )

  METHOD cba_Item.
    lcl_buffer=>lt_create = VALUE #( FOR wa_hdr IN lcl_buffer=>lt_hdr
                                     FOR entity IN entities_cba WHERE ( %pid = wa_hdr-pid_hdr )
                                    (
                                       cid_hdr = wa_hdr-cid_hdr
                                       pid_hdr = wa_hdr-pid_hdr
                                         salesdocument                = wa_hdr-salesdocument
                                        salesdocumenttype            = wa_hdr-salesdocumenttype
                                        creationdate                 = wa_hdr-creationdate
                                        creationtime                 = wa_hdr-creationtime
                                        createdbyuser                = wa_hdr-createdbyuser
                                        transactioncurrency          =  wa_hdr-transactioncurrency
                                        totalnetamount               = wa_hdr-totalnetamount
                                        salesorganization            = wa_hdr-salesorganization
                                        distributionchannel          = wa_hdr-distributionchannel
                                        organizationdivision         = wa_hdr-organizationdivision
                                        soldtoparty                  = wa_hdr-soldtoparty
                                        customerpaymentterms         = wa_hdr-customerpaymentterms
                                        item = VALUE #( FOR item IN entity-%target
                                                        (
                                                        CORRESPONDING #(  item MAPPING cid_item = %cid )
                                                        )
                                                      )
                                      ) ).

    mapped-item = VALUE #( FOR entity IN entities_cba
                           FOR item IN entity-%target
                           (  %cid = item-%cid
                              %pid = item-SalesDocumentItem
                              SalesDocument = item-SalesDocument
                              SalesDocumentItem = item-SalesDocumentItem
                             ) ).
  ENDMETHOD.

  METHOD so_count_status.
    DATA(lv_kunnr) = keys[ 1 ]-%param-Customer.
    lv_kunnr = |{ lv_kunnr ALPHA = IN } |.

    SELECT
    COUNT( CASE gbstk WHEN 'A' THEN gbstk  END ) AS notprocessed,
    COUNT( CASE gbstk WHEN 'B' THEN gbstk  END ) AS partiallyprocessed,
    COUNT( CASE gbstk WHEN 'C' THEN gbstk  END ) AS completelyprocessed,
    COUNT( CASE gbstk WHEN ' ' THEN gbstk  END ) AS notrelevant
    FROM vbak
    WHERE kunnr = @lv_kunnr
    INTO ( @DATA(lv_notprocessed), @DATA(lv_partiallyprocessed),
    @DATA(lv_completelyprocessed), @DATA(lv_notrelevant) ).

    result = VALUE #( ( %cid = keys[ 1 ]-%cid
                         %param = VALUE #( Customer = lv_kunnr
                                           CompletelyProcessed = lv_completelyprocessed
                                           NotProcessed = lv_notprocessed
                                           NotRelevant = lv_notrelevant
                                           PartiallyProcessed = lv_partiallyprocessed
                                          )
                   ) ).
  ENDMETHOD.

ENDCLASS.
